## Код DAG

```python

# ------------------------------------------------------------------
# 1. ИМПОРТ НЕОБХОДИМЫХ МОДУЛЕЙ
# ------------------------------------------------------------------

from datetime import datetime          # Модуль для работы с датой и временем
from airflow import DAG                # Модуль для создания DAG (графа задач)
from airflow.operators.python import PythonOperator   # Оператор для выполнения Python-функций
from airflow.operators.dummy import DummyOperator     # Оператор-заглушка (ничего не делает)
import json                            # Модуль для сохранения данных в JSON-формате
import os                              # Модуль для работы с файловой системой


# ------------------------------------------------------------------
# 2. НАСТРОЙКА ДИРЕКТОРИИ ДЛЯ ХРАНЕНИЯ ДАННЫХ
# ------------------------------------------------------------------

# Определяем путь к папке, где будут сохраняться данные
# ~/airflow_home/data - папка в домашней директории пользователя
DATA_DIR = os.path.expanduser("~/airflow_home/data")

# Создаем папку, если она еще не существует
# exist_ok=True - не выдавать ошибку, если папка уже есть
os.makedirs(DATA_DIR, exist_ok=True)


# ------------------------------------------------------------------
# 3. ЗАДАЧА 1: ГЕНЕРАЦИЯ БИЛЕТОВ
# ------------------------------------------------------------------

def generate():
    """
    Функция генерирует 100 тестовых билетов и сохраняет их в JSON-файл.
    Каждый билет содержит:
    - id: номер билета (от 0 до 99)
    - movie: название фильма (Avatar, Oppenheimer или Barbie по очереди)
    - price: цена билета (400, 450 или 500 рублей)
    """
    
    tickets = []  # Пустой список для хранения билетов
    
    # Цикл для создания 100 билетов (от 0 до 99)
    for i in range(100):
        tickets.append({
            "id": i,                                                   # номер билета
            "movie": ["Avatar", "Oppenheimer", "Barbie"][i % 3],       # фильм по очереди
            "price": 400 + (i % 3) * 50                                # цена: 400, 450 или 500
        })
    
    # Сохраняем список билетов в файл tickets.json
    with open(f"{DATA_DIR}/tickets.json", "w") as f:
        json.dump(tickets, f)      # запись в JSON-формате
    
    # Выводим информацию в консоль
    print(f"Создано {len(tickets)} билетов")


# ------------------------------------------------------------------
# 4. ЗАДАЧА 2: РАСЧЕТ КАССОВЫХ СБОРОВ
# ------------------------------------------------------------------

def report():
    """
    Функция читает файл с билетами, суммирует цены и выводит общую выручку.
    """
    
    # Открываем файл tickets.json и читаем данные
    with open(f"{DATA_DIR}/tickets.json", "r") as f:
        tickets = json.load(f)     # загрузка данных из JSON
    
    # Суммируем цены всех билетов
    total = sum(t["price"] for t in tickets)
    
    # Выводим общую выручку в консоль
    print(f"Общая выручка: {total} рублей")


# ------------------------------------------------------------------
# 5. СОЗДАНИЕ DAG (ГРАФА ЗАДАЧ)
# ------------------------------------------------------------------

dag = DAG(
    'test_cinema',                      # Уникальное имя DAG
    start_date=datetime(2024, 1, 1),    # Дата начала работы DAG
    schedule_interval=None,             # Расписание: None = только ручной запуск
    catchup=False                       # Не выполнять пропущенные запуски
)


# ------------------------------------------------------------------
# 6. ОПРЕДЕЛЕНИЕ ЗАДАЧ (ОПЕРАТОРОВ)
# ------------------------------------------------------------------

# start - начальная точка (ничего не делает, нужна для структуры)
start = DummyOperator(task_id='start', dag=dag)

# task1 - задача генерации билетов (вызывает функцию generate)
task1 = PythonOperator(
    task_id='generate_data',            # Имя задачи
    python_callable=generate,           # Функция, которую нужно выполнить
    dag=dag                             # Привязка к DAG
)

# task2 - задача расчета выручки (вызывает функцию report)
task2 = PythonOperator(
    task_id='make_report',              # Имя задачи
    python_callable=report,             # Функция, которую нужно выполнить
    dag=dag                             # Привязка к DAG
)

# end - конечная точка (ничего не делает, нужна для структуры)
end = DummyOperator(task_id='end', dag=dag)


# ------------------------------------------------------------------
# 7. ОПРЕДЕЛЕНИЕ ПОРЯДКА ВЫПОЛНЕНИЯ ЗАДАЧ
# ------------------------------------------------------------------

# Оператор >> задает последовательность выполнения
# start → task1 → task2 → end
start >> task1 >> task2 >> end


# ------------------------------------------------------------------
# 8. РЕЗУЛЬТАТ РАБОТЫ DAG
# ------------------------------------------------------------------

# 1. В папке ~/airflow_home/data создается файл tickets.json
# 2. В файле содержится 100 билетов с фильмами и ценами
# 3. В консоль выводится общая выручка (пример: 44950 рублей)
# ------------------------------------------------------------------
